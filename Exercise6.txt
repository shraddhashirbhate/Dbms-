1. Display all the Suppliers with the same Status as the supplier, ‘CLARK’.
Ans->
	select * from supplier
	where status = (select status from supplier 
	where Sname = 'Clark');

--------------------------------------------------------------------------------------------------------------------------------------------------
2. Display all the Employees in the same department as the employee ‘MILLER’.
Ans->
	select * from emp 
	where deptno = (select deptno from emp 
	where ename = 'Miller');
                
--------------------------------------------------------------------------------------------------------------------------------------------------
3. Display all the Parts which have more Weight than all the Red parts.
Ans->
	select Pname from parts 
	where weight >Any (select weight from parts
	where Color = 'Red');

5. Display all the Parts with Weight less than all the Green parts.
Ans->

6. Display the name of the Supplier who has sold the maximum Quantity (in one
sale).
Ans->
	select Sname from supplier ,SUPPLIER_PARTS_PROJECT
	where SUPPLIER_PARTS_PROJECT.`S#` = supplier .`S#` 
	and SUPPLIER_PARTS_PROJECT.Qty >Any (select max(Qty) from (Select max(Qty) , 
	`S#` from SUPPLIER_PARTS_PROJECT
									 group by `S#` ) as temp );

--------------------------------------------------------------------------------------------------------------------------------------------------
7. Display the name of the Employee with the minimum Salary.
Ans->

	select ename , min(sal) from emp
	group by ename 
	order by min(sal) 
	limit 1 ; 


--------------------------------------------------------------------------------------------------------------------------------------------------
8. Display the name of the Supplier who has sold the maximum overall Quantity (sum
of Sales).
Ans->
	select Sname from supplier ,SUPPLIER_PARTS_PROJECT
	where SUPPLIER_PARTS_PROJECT.`S#` = supplier .`S#` 
	and SUPPLIER_PARTS_PROJECT.Qty = (select max(Qty) from (Select max(Qty) , 
	`S#` from SUPPLIER_PARTS_PROJECT
									 group by `S#` order by max(Qty) desc limit 1  ) as temp );

--------------------------------------------------------------------------------------------------------------------------------------------------
9. Display the name of the Department with the maximum number of Employees.
Ans->
	select dept.deptname , count(emp.ename) from emp , dept 
	where emp.deptno = dept.deptno
	group by dept.deptname
	order by count(emp.ename) desc limit 1 ;
